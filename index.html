// ========================================== // 核心設定：請確認您的 Cloudflare Worker 網址 // ========================================== const MY_WORKER_URL = "https://taichung-parking-api.tmya-w-l-c3e7b8e0-5b30-41a8.workers.dev/"; let allParkingSpots = []; // 儲存後端抓回來的所有車位資料 let backupSpots = []; // 備用核心資料庫 // 網頁初始化載入 document.addEventListener("DOMContentLoaded", () => { // 1. 自動向 Worker 獲取最新即時資料 fetchTDXParkingData(); // 2. 自動強力綁定：讓輸入框支援按下 Enter 鍵也能觸發智慧搜尋 const searchInput = document.getElementById('search-input'); if (searchInput) { searchInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') { doAdvancedSearch(); } }); } // 3. 自動強力綁定：點擊搜尋按鈕時觸發智慧搜尋 const searchBtn = document.getElementById('search-btn'); if (searchBtn) { searchBtn.removeAttribute('onclick'); // 移除舊的 onclick 避免衝突 searchBtn.addEventListener('click', doAdvancedSearch); } }); // 向您的 Cloudflare Worker 獲取最新即時資料（含預測資料與收費資訊） async function fetchTDXParkingData() { try { if (MY_WORKER_URL.includes("您的帳號")) { throw new Error("尚未設定正確的 Worker 網址"); } const dataResponse = await fetch(MY_WORKER_URL); if (!dataResponse.ok) throw new Error(`雲端通道回應異常 (狀態碼:${dataResponse.status})`); const tdxRealData = await dataResponse.json(); if (tdxRealData && tdxRealData.error) { throw new Error(tdxRealData.error); } let availList = []; if (tdxRealData && tdxRealData.ParkingAvailabilities) { availList = tdxRealData.ParkingAvailabilities; } else if (Array.isArray(tdxRealData)) { availList = tdxRealData; } if (availList && Array.isArray(availList) && availList.length > 0) { allParkingSpots = availList.map(item => { const targetName = item.POSITION || (item.CarParkName && item.CarParkName.Zh_tw) || "公有停車場"; return { POSITION: targetName, ADDRESS: item.ADDRESS || item.CarParkAddress || item.Address || "台中市區域", TOTAL: parseInt(item.TOTAL || item.TotalSpaces) || 100, AVAILABLE: item.AVAILABLE !== undefined ? item.AVAILABLE : (item.AvailableSpaces !== undefined ? item.AvailableSpaces : 0), IS_PREDICTED: item.IS_PREDICTED === true, FARE_DESC: item.FARE_DESC || "依現場公告費率為準", // 🌟 確實接收後端收費敘述 LATITUDE: parseFloat(item.LATITUDE || item.Latitude) || 24.1372, LONGITUDE: parseFloat(item.LONGITUDE || item.Longitude) || 120.6856 }; }); allParkingSpots.sort((a, b) => b.AVAILABLE - a.AVAILABLE); document.getElementById('status-text').innerText = `🟢 同步成功！已透過雲端通道載入 ${allParkingSpots.length} 個即時車位。`; renderParkingList(allParkingSpots.slice(0, 15)); } else { throw new Error('雲端回傳格式不符，未偵測到車位陣列'); } } catch (e) { console.log("轉向安全備用模式:", e); allParkingSpots = [...backupSpots]; document.getElementById('status-text').innerText = `🟡 目前運行於核心備用模式 (原因:${e.message})`; renderParkingList(allParkingSpots); } } // 計算經緯度公里距離公式 (Haversine 公式) function calculateDistance(lat1, lon1, lat2, lon2) { const R = 6371e3; const phi1 = lat1 * Math.PI / 180; const phi2 = lat2 * Math.PI / 180; const deltaPhi = (lat2 - lat1) * Math.PI / 180; const deltaLambda = (lon2 - lon1) * Math.PI / 180; const a = Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) + Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)); return R * c; } // 智慧核心：地名轉座標 ＋ 半徑範圍雙軌搜尋 async function doAdvancedSearch() { const searchInput = document.getElementById('search-input'); if (!searchInput) return; const keyword = searchInput.value.trim(); if (!keyword) { renderParkingList(allParkingSpots.slice(0, 15)); return; } document.getElementById('status-text').innerText = `🔍 正在計算 「${keyword}」 的周邊車位與距離...`; try { // 自動加上 "台中" 關鍵字，確保 OpenStreetMap 能精準定位在台中市內 const queryText = keyword.includes("台中") ? keyword : `台中 ${keyword}`; const geoUrl = `https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(queryText)}&limit=1`; const geoResponse = await fetch(geoUrl, { headers: { 'Accept-Language': 'zh-TW', 'User-Agent': 'TaichungParkingApp/1.0' } }); const geoData = await geoResponse.json(); let targetLat = null; let targetLon = null; if (geoData && geoData.length > 0) { targetLat = parseFloat(geoData[0].lat); targetLon = parseFloat(geoData[0].lon); } let searchResults = allParkingSpots.map(spot => { let distance = null; let matchText = 0; if (targetLat !== null && targetLon !== null) { distance = calculateDistance(targetLat, targetLon, spot.LATITUDE, spot.LONGITUDE); } if (spot.POSITION.toLowerCase().includes(keyword.toLowerCase()) || spot.ADDRESS.toLowerCase().includes(keyword.toLowerCase())) { matchText = 1; } // 🌟 核心關鍵：將包含 FARE_DESC 在內的所有原始欄位完整擴充保留，絕不漏接！ return { ...spot, DISTANCE: distance, MATCH_TEXT: matchText }; }); // 篩選：名字符合關鍵字，或是距離目標地標 3.5 公里以內的所有停車場 let filteredSpots = searchResults.filter(spot => { return spot.MATCH_TEXT === 1 || (spot.DISTANCE !== null && spot.DISTANCE <= 3500); }); // 排序：有計算出距離的，由近到遠排；其餘依照剩餘車位從多到少排 filteredSpots.sort((a, b) => { if (a.DISTANCE !== null && b.DISTANCE !== null) { return a.DISTANCE - b.DISTANCE; } return b.AVAILABLE - a.AVAILABLE; }); renderParkingList(filteredSpots); document.getElementById('status-text').innerText = `🟢 搜尋完成！已列出「${keyword}」周邊最佳車位。`; } catch (error) { console.error("地理編碼異常，自動降級為純文字安全模糊搜尋", error); const fallback = allParkingSpots.filter(spot => spot.POSITION.includes(keyword) || spot.ADDRESS.includes(keyword) ); renderParkingList(fallback); } } // 將資料渲染至網頁卡片面版 function renderParkingList(spots) { const listContainer = document.getElementById('parking-list'); listContainer.innerHTML = ""; if (!spots || spots.length === 0) { listContainer.innerHTML = `
找不到相符的停車場，換個名稱試試吧 😢
`; return; } spots.forEach(spot => { let countColor = "text-green-600 font-black"; let bgColor = "border-gray-200"; let fontStyle = ""; if (spot.IS_PREDICTED === true) { countColor = "text-amber-500 font-extrabold"; fontStyle = "italic"; } else { if (spot.AVAILABLE <= 0) { countColor = "text-red-500 font-black"; bgColor = "bg-red-50 border-red-100"; } else if (spot.AVAILABLE < 10) { countColor = "text-yellow-600 font-black"; } } // 計算並呈現精緻的距離提醒標籤 let distanceBadge = ""; if (spot.DISTANCE !== null && spot.DISTANCE !== undefined) { const distKm = spot.DISTANCE / 1000; const distText = spot.DISTANCE < 1000 ? `${Math.round(spot.DISTANCE)} 公尺` : `${distKm.toFixed(1)} 公里`; distanceBadge = ` 距離約 ${distText}`; } const sourceBadge = spot.IS_PREDICTED === true ? `🤖 大數據預估` : `⚡ 官方即時連線`; const cardHtml = `

${spot.POSITION}

${spot.ADDRESS}

收費：${spot.FARE_DESC}

總車位: ${spot.TOTAL} ${sourceBadge} ${distanceBadge}
剩餘車位
${spot.AVAILABLE}
`; listContainer.innerHTML += cardHtml; }); } // 導航函數 function navigate(lat, lon) { if (navigator.userAgent.match(/(iPhone|iPad|iPod|Macintosh)/i)) { window.open(`maps://maps.apple.com/?daddr=${lat},${lon}&dirflg=d`); } else { window.open(`https://www.google.com/maps/dir/?api=1&destination=${lat},${lon}`); } }
